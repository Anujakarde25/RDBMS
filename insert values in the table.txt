create table student(rno int,
sname char(30),
per float,
class varchar(30));

DECLARE
rno int;
sname char(30);
per float;
class varchar(30);
BEGIN
rno:=&rno;
sname:='&sname';
per:=&per;

If per>=75 THEN
class:='distinction';
ELSE IF per>=65 THEN
class:='first class';
ELSE IF per>=55 THEN
class:='second class';
ELSE If per>=35 THEN
class:='pass';
ELSE
class:='Fail';
END IF;
END IF;
END IF;
END IF;

insert into student values(rno,sname,per,class);

END;
/
